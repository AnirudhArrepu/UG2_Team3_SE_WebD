   const [isChatOpen, setIsChatOpen] = useState(false);

   const toggleChat = () => {
      setIsChatOpen(prevState => !prevState);
   };


   return(
    <>
        <button className="chat-icon" onClick={toggleChat}>
            <FaComment size={30} color="#007BFF" />
        </button>

        {isChatOpen && <Chat toggleChat={toggleChat} />}
    <>
   )
